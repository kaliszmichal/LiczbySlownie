Sub ZamienKwotyNaSlownie()
    Dim docText As String
    Dim regEx As Object
    Dim matches As Object
    Dim match As Object
    Dim fullMatch As String
    Dim replacement As String
    Dim processed As Object
    Dim rng As Range
    Dim zakresRoboczy As Range
    Dim czyCalyDokument As Boolean
    Dim czyCosZamieniono As Boolean ' Flaga blokująca nakładanie się Części 4
    
    ' Zmienne dla walut
    Dim kwota As Variant, grosze As Variant, waluta As Variant
    Dim slownieWaluta As String, walutaFormat As String, groszeTekst As String
    
    ' Zmienne dla powierzchni
    Dim powCalkowita As String, powUlamek As String, slowniePowierzchnia As String
    
    ' Zmienne dla ułamków
    Dim przedChar As String
    Dim licznik As Variant, mianownik As Variant
    Dim slownieLicznik As String, slownieMianownik As String
    
    ' OKREŚLENIE ZAKRESU DZIAŁANIA:
    If Selection.Start = Selection.End Then
        Set zakresRoboczy = ActiveDocument.Range
        czyCalyDokument = True
    Else
        Set zakresRoboczy = Selection.Range
        czyCalyDokument = False
    End If
    
    Application.ScreenUpdating = False
    Set regEx = CreateObject("VBScript.RegExp")
    regEx.Global = True
    regEx.IgnoreCase = True
    czyCosZamieniono = False
    
    ' --- CZĘŚĆ 1: KWOTY WALUTOWE (PLN, EUR, euro, zł, zł., $) ---
    docText = zakresRoboczy.Text
    ' POPRAWKA: Dodano \$ do listy wyszukiwanych walut
    regEx.Pattern = "\b([0-9][0-9 \xA0]*)(?:,([0-9]{2}))?\s*(PLN|EUR|euro|zł\.?|\$)(?![a-zA-Z0-9ąęćłńóśźżĄĘĆŁŃÓŚŹŻ])(?!\s*\(\s*słownie)"
    Set matches = regEx.Execute(docText)
    Set processed = CreateObject("Scripting.Dictionary")
    
    For Each match In matches
        fullMatch = match.Value
        If Not processed.Exists(fullMatch) Then
            processed.Add fullMatch, True
            
            kwota = Replace(match.SubMatches(0), " ", "")
            kwota = Replace(kwota, Chr(160), "")
            
            If Not IsEmpty(match.SubMatches(1)) Then
                grosze = match.SubMatches(1)
            Else
                grosze = ""
            End If
            
            waluta = LCase(match.SubMatches(2))
            slownieWaluta = LiczbaSlownie(kwota)
            
            ' Rozpoznawanie waluty i ustalanie poprawnej formy gramatycznej
            If waluta = "pln" Or waluta = "zł" Or waluta = "zł." Then
                walutaFormat = "złotych"
            ElseIf waluta = "eur" Or waluta = "euro" Then
                walutaFormat = "euro"
            ElseIf waluta = "$" Then
                ' Odmiana słowa "dolar"
                walutaFormat = SuffixDolara(kwota)
            Else
                walutaFormat = waluta
            End If
            
            ' Formatowanie groszy/centów
            If grosze <> "" And grosze <> "00" Then
                groszeTekst = " i " & grosze & "/100"
            Else
                groszeTekst = ""
            End If
            
            replacement = fullMatch & " (słownie " & walutaFormat & ": " & slownieWaluta & groszeTekst & ")"
            
            Set rng = zakresRoboczy.Duplicate
            With rng.Find
                .ClearFormatting
                .Text = fullMatch
                .replacement.Text = replacement
                .Forward = True
                .Wrap = 0
                .Format = False
                .MatchCase = False
                .MatchWholeWord = False
                .MatchWildcards = False
                .Execute Replace:=2
            End With
            czyCosZamieniono = True
        End If
    Next match
    
    ' --- CZĘŚĆ 2: MIARY POWIERZCHNI (m2 lub m²) ---
    docText = zakresRoboczy.Text
    regEx.Pattern = "\b([0-9][0-9 \xA0]*)(?:,([0-9]+))?\s*(m2|m²)(?![a-zA-Z0-9ąęćłńóśźżĄĘĆŁŃÓŚŹŻ])(?!\s*\(\s*[^)]*metr)"
    Set matches = regEx.Execute(docText)
    Set processed = CreateObject("Scripting.Dictionary")
    
    For Each match In matches
        fullMatch = match.Value
        If Not processed.Exists(fullMatch) Then
            processed.Add fullMatch, True
            
            powCalkowita = Replace(match.SubMatches(0), " ", "")
            powCalkowita = Replace(powCalkowita, Chr(160), "")
            
            If Not IsEmpty(match.SubMatches(1)) Then
                powUlamek = match.SubMatches(1)
            Else
                powUlamek = ""
            End If
            
            slowniePowierzchnia = PowierzchniaSlownie(powCalkowita, powUlamek)
            replacement = fullMatch & " (" & slowniePowierzchnia & ")"
            
            Set rng = zakresRoboczy.Duplicate
            With rng.Find
                .ClearFormatting
                .Text = fullMatch
                .replacement.Text = replacement
                .Forward = True
                .Wrap = 0
                .Format = False
                .MatchCase = False
                .MatchWholeWord = False
                .MatchWildcards = False
                .Execute Replace:=2
            End With
            czyCosZamieniono = True
        End If
    Next match
    
    ' --- CZĘŚĆ 3: UŁAMKI ZWYKŁE (X/Y) ---
    docText = zakresRoboczy.Text
    regEx.Pattern = "\b([0-9]+)/([0-9]+)\b(?!\s*/)(?!\s*\(\s*[^)]*łamane)"
    Set matches = regEx.Execute(docText)
    Set processed = CreateObject("Scripting.Dictionary")
    
    For Each match In matches
        fullMatch = match.Value
        
        przedChar = ""
        If match.FirstIndex > 0 Then
            przedChar = Mid(docText, match.FirstIndex, 1)
        End If
        
        Dim jestGroszem As Boolean
        jestGroszem = False
        If match.FirstIndex >= 3 Then
            Dim startPoz As Long
            If match.FirstIndex - 10 > 1 Then
                startPoz = match.FirstIndex - 10
            Else
                startPoz = 1
            End If
            
            Dim kontekstPrzed As String
            kontekstPrzed = Mid(docText, startPoz, 11)
            If InStr(LCase(kontekstPrzed), " i ") > 0 Or InStr(LCase(kontekstPrzed), "słownie") > 0 Or InStr(LCase(kontekstPrzed), "metr") > 0 Then
                jestGroszem = True
            End If
        End If
        
        If przedChar <> "/" And Not jestGroszem Then
            If Not processed.Exists(fullMatch) Then
                processed.Add fullMatch, True
                
                licznik = match.SubMatches(0)
                mianownik = match.SubMatches(1)
                
                slownieLicznik = LiczbaSlownie(licznik)
                slownieMianownik = LiczbaSlownie(mianownik)
                
                replacement = fullMatch & " (" & slownieLicznik & " łamane przez " & slownieMianownik & ")"
                
                Set rng = zakresRoboczy.Duplicate
                With rng.Find
                    .ClearFormatting
                    .Text = fullMatch
                    .replacement.Text = replacement
                    .Forward = True
                    .Wrap = 0
                    .Format = False
                    .MatchCase = False
                    .MatchWholeWord = False
                    .MatchWildcards = False
                    .Execute Replace:=2
                End With
                czyCosZamieniono = True
            End If
        End If
    Next match
    
    ' --- CZĘŚĆ 4: SAME LICZBY CAŁKOWITE (Działa tylko przy ZAZNACZENIU tekstu) ---
    If Not czyCalyDokument And Not czyCosZamieniono Then
        docText = zakresRoboczy.Text
        regEx.Pattern = "\b([0-9][0-9 \xA0]*)\b(?![a-zA-Z0-9ąęćłńóśźżĄĘĆŁŃÓŚŹŻ])(?!\s*\(\s*słownie)"
        Set matches = regEx.Execute(docText)
        Set processed = CreateObject("Scripting.Dictionary")
        
        For Each match In matches
            fullMatch = match.Value
            If Not processed.Exists(fullMatch) Then
                processed.Add fullMatch, True
                
                Dim czystaLiczba As String
                czystaLiczba = Replace(fullMatch, " ", "")
                czystaLiczba = Replace(czystaLiczba, Chr(160), "")
                
                If IsNumeric(czystaLiczba) Then
                    replacement = fullMatch & " (słownie: " & LiczbaSlownie(czystaLiczba) & ")"
                    
                    Set rng = zakresRoboczy.Duplicate
                    With rng.Find
                        .ClearFormatting
                        .Text = fullMatch
                        .replacement.Text = replacement
                        .Forward = True
                        .Wrap = 0
                        .Format = False
                        .MatchCase = False
                        .MatchWholeWord = False
                        .MatchWildcards = False
                        .Execute Replace:=2
                    End With
                End If
            End If
        Next match
    End If
    
    Application.ScreenUpdating = True
End Sub

' NOWA POMOCNICZA FUNKCJA: Odmienia słowo "dolar" na podstawie kwoty
Function SuffixDolara(ByVal liczba As Variant) As String
    Dim n As String
    n = Replace(CStr(liczba), " ", "")
    n = Replace(n, Chr(160), "")
    
    If n = "1" Then
        SuffixDolara = "dolar"
        Exit Function
    End If
    
    Dim lenN As Integer: lenN = Len(n)
    Dim ostatnie2 As Integer, ostatnia As Integer
    
    If lenN >= 2 Then
        ostatnie2 = CInt(Right(n, 2))
    Else
        ostatnie2 = CInt(n)
    End If
    ostatnia = CInt(Right(n, 1))
    
    If ostatnie2 >= 12 And ostatnie2 <= 14 Then
        SuffixDolara = "dolarów"
    ElseIf ostatnia >= 2 And ostatnia <= 4 Then
        SuffixDolara = "dolary"
    Else
        SuffixDolara = "dolarów"
    End If
End Function

Function PowierzchniaSlownie(ByVal czescCalkowita As String, ByVal czescUlamkowa As String) As String
    Dim wynik As String
    Dim slownieCala As String
    Dim suf As String
    slownieCala = LiczbaSlownie(czescCalkowita)
    suf = SuffixMetra(czescCalkowita)
    wynik = slownieCala & " " & suf
    If czescUlamkowa <> "" And Val(czescUlamkowa) > 0 Then
        If Len(czescUlamkowa) = 1 Then czescUlamkowa = czescUlamkowa & "0"
        If Len(czescUlamkowa) > 2 Then czescUlamkowa = Left(czescUlamkowa, 2)
        Dim slownieUlamek As String, sufSetnych As String, ulaNum As Integer
        ulaNum = CInt(czescUlamkowa)
        slownieUlamek = LiczbaSlownie(czescUlamkowa)
        If ulaNum = 1 Then
            sufSetnych = "jedna setna" 
            wynik = wynik & " i " & czescUlamkowa & "/100 " & sufSetnych
        Else
            Dim ost2 As Integer, ost1 As Integer
            ost2 = ulaNum Mod 100
            ost1 = ulaNum Mod 10
            If ost2 >= 12 And ost2 <= 14 Then
                sufSetnych = "setnych"
            ElseIf ost1 >= 2 And ost1 <= 4 Then
                sufSetnych = "setne"
            Else
                sufSetnych = "setnych"
            End If
            wynik = wynik & " i " & czescUlamkowa & "/100 " & slownieUlamek & " " & sufSetnych
        End If
    End If
    PowierzchniaSlownie = wynik
End Function

Function SuffixMetra(ByVal liczba As Variant) As String
    Dim n As String
    n = Replace(CStr(liczba), " ", "")
    n = Replace(n, Chr(160), "")
    
    If n = "1" Then
        SuffixMetra = "metr kwadratowy"
        Exit Function
    End If
    
    Dim lenN As Integer: lenN = Len(n)
    Dim ostatnie2 As Integer, ostatnia As Integer
    
    If lenN >= 2 Then
        ostatnie2 = CInt(Right(n, 2))
    Else
        ostatnie2 = CInt(n)
    End If
    
    ostatnia = CInt(Right(n, 1))
    
    If ostatnie2 >= 12 And ostatnie2 <= 14 Then
        SuffixMetra = "metrów kwadratowych"
    ElseIf ostatnia >= 2 And ostatnia <= 4 Then
        SuffixMetra = "metry kwadratowe"
    Else
        SuffixMetra = "metrów kwadratowych"
    End If
End Function

Function LiczbaSlownie(ByVal liczba As Variant) As String
    Dim jednosci As Variant, dziesiatki As Variant, setki As Variant, nascie As Variant
    Dim potegi(3, 3) As String, strLiczba As String
    strLiczba = Replace(CStr(liczba), " ", "")
    strLiczba = Replace(strLiczba, Chr(160), "")
    jednosci = Array("", "jeden", "dwa", "trzy", "cztery", "pięć", "sześć", "siedem", "osiem", "dziewięć")
    nascie = Array("dziesięć", "jedenaście", "dwanaście", "trzynaście", "czternaście", "piętnaście", "szesnaście", "siedemnaście", "osiemnaście", "dziewiętnaście")
    dziesiatki = Array("", "", "dwadzieścia", "trzydzieści", "czterdzieści", "pięćdziesiąt", "sześćdziesiąt", "siedemdziesiąt", "osiemdziesiąt", "dziewięćdziesiąt")
    setki = Array("", "sto", "dwieście", "trzysta", "czterysta", "pięćset", "sześćset", "siedemset", "osiemset", "dziewięćset")
    potegi(1, 0) = "tysiąc":  potegi(1, 1) = "tysiące":  potegi(1, 2) = "tysięcy"
    potegi(2, 0) = "milion":  potegi(2, 1) = "miliony":  potegi(2, 2) = "milionów"
    potegi(3, 0) = "miliard": potegi(3, 1) = "miliardy": potegi(3, 2) = "miliardów"
    Dim wynik As String, i As Integer, dlugosc As Integer, triada As String, temp As String
    Dim setka As Integer, dziesiatka As Integer, jednosc As Integer, forma As Integer, j_calkowita As Integer
    
    If strLiczba = "0" Or strLiczba = "" Then
        LiczbaSlownie = "zero"
        Exit Function
    End If
    
    dlugosc = Len(strLiczba)
    If dlugosc Mod 3 > 0 Then
        strLiczba = String(3 - (dlugosc Mod 3), "0") & strLiczba
    End If
    
    dlugosc = Len(strLiczba)
    Dim g As Integer: g = (dlugosc \ 3) - 1
    
    For i = 1 To dlugosc Step 3
        triada = Mid(strLiczba, i, 3)
        setka = CInt(Mid(triada, 1, 1))
        dziesiatka = CInt(Mid(triada, 2, 1))
        jednosc = CInt(Mid(triada, 3, 1))
        j_calkowita = CInt(triada)
        
        If j_calkowita > 0 Then
            temp = ""
            If setka > 0 Then temp = temp & setki(setka) & " "
            
            If dziesiatka = 1 Then
                temp = temp & nascie(jednosc) & " "
            Else
                If dziesiatka > 1 Then temp = temp & dziesiatki(dziesiatka) & " "
                If jednosc > 0 Then temp = temp & jednosci(jednosc) & " "
            End If
            
            If j_calkowita = 1 Then
                forma = 0
            ElseIf (dziesiatka <> 1) And (jednosc >= 2 And jednosc <= 4) Then
                forma = 1
            Else
                forma = 2
            End If
            wynik = wynik & temp & potegi(g, forma) & " "
        End If
        g = g - 1
    Next i
    LiczbaSlownie = Trim(wynik)
End Function
